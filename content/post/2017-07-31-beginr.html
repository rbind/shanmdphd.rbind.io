x <- 1:10 
y <- 1:10 + rnorm(10)

beginr::plotlm(x, y, xlim = range(as.numeric(x), na.rm = TRUE), 
       ylim = range(as.numeric(y), na.rm = TRUE), plot.title = "linear regression", xlab = "x", ylab = "y", 
       refline = TRUE, slope = 1, intercept = 0, showr2 = TRUE, showleg = TRUE)
