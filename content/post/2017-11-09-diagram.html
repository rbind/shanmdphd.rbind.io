이 글1에서 배웠습니다.

library(DiagrammeR)

seq <- grViz('digraph{c->a; c->b; c->d;}')

DiagrammeRsvg::export_svg(seq) %>% 
  charToRaw %>% 
  rsvg::rsvg_png(file = 'seq.png', width = 1000)
knitr::include_graphics('seq.png')
